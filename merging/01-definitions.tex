\begin{definitions}
	\definition{Вычислительный узел (узел)}{устройство, выполняющее основную логику обработки запроса \cite{com_analysis}.}
	\definition{Хеш-таблица}{структура данных для реализации ассоциативных массиов \cite{hash}.}
	\definition{NAT}{Network Address Translation, метод трансляции сетевых адресов~\cite{nat}.}
	\definition{IP-адрес}{уникальный адрес, идентифицирующий устройство в интернете или локальной сети~\cite{ip-def}.}
	\definition{URL-адрес}{Uniform Resource Locator, адрес, который выдан уникальному ресурсу в интернете~\cite{url-def}.}
\end{definitions}
